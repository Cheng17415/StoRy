BBDD - Postgres

Usuario
- id (PK)
- name
- email (UNIQUE)
- username (UNIQUE)
- password
- status        -- ACTIVO, BLOQUEADO, ELIMINADO
- provider      -- LOCAL / GOOGLE
- provider_id   -- id de Google (opcional)
- fecha_registro
- fecha_ultimo_login


Perfil
	-id
	-perfil

UsuarioPerfil
	-idUsuario (PK, FK)
	-idPerfil (PK,FK)

Producto
- id (PK)
- nombre
- descripcion
- codigo      -- QR o Barcode (UNIQUE)
- precio
- cantidad
- stock_minimo
- activo
- fecha_creacion
- fecha_actualizacion
- imagen
-categoria_id(FK)

Categoria
-id(PK)
-nombre
-descripción

MovimientoStock
- id (PK)
- producto_id (FK)
- tipo        -- ENTRADA / SALIDA / AJUSTE
- cantidad
- fecha
- usuario_id (FK)
- observacion
